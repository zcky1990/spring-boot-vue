<script>

export default {
  name: 'bootstrap',
 
  data () {

      showResponse: false,
      response: '',
      fullResponse: {
        config: {
          foo: '',
          bar: ''
        }

    callRestService () {
      AXIOS.get(`/hello`)
        .then(response => {

          this.headers = response.headers
